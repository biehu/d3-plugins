YXChart.js

说明文档

怎么使用

引入jQuery

<div id="conversionRateLineId" style="height: 300px;"></div>
<script src="../lib/jquery.min.js"></script>

引入d3或者echarts

<script src="../lib/d3.v3.min.js"></script> 或者				
<script src="../lib/echarts-all.js"></script>

引入d3类库或者配置类

<script src="../src/YXCharts.js"></script>
<script src="../src/config/rateLine.js"></script>

调用 new YXChart

var yxcharts = new YXCharts(
	 document.getElementById('conversionRateLineId'),
	 'rateLine',
	 {
	    xAxis: [
	       {
	           data: ["5月", "6月", "7月", "8月", "9月", "10月"]
	       }
	    ],
	    series: [
	       {
	           data: [0, "0.40", "0.32", "0.49", "0.60", "0.19"]
	       }
	    ]
	 },
	 {
	    yAxis: [
	        {
	            axisLabel: {
	                textStyle: {
	                    color: '#f00'
	                }
	            }
	        }
	    ]
	}
);
